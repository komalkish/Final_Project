-- Creating tables for Life Expectancy Project
CREATE TABLE who (
     country VARCHAR NOT NULL,
     life_exp int NOT NULL,
     PRIMARY KEY (country),
     UNIQUE (life_exp)
);

CREATE TABLE suicide (
	country VARCHAR NOT NULL,
	year INT NOT NULL,
	suicide_no INT NOT NULL,
	pop INT NOT NULL,
	gdp INT NOT NULL,
	PRIMARY KEY (country)
);

CREATE TABLE Lifeexp_suicide (
country VARCHAR NOT NULL,
	year INT NOT NULL,
	suicide_no INT NOT NULL,
FOREIGN KEY (year) REFERENCES suicide (year),
	PRIMARY KEY (country, country)
);
